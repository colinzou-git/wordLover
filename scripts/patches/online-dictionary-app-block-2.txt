async function lookupTermWithOnlineFallback(input, allowOnline = false) {
  const primary = await lookupTermWithFullFallback(input);
  if (primary.status !== "not_found" || isChineseInput(input) || !allowOnline) return primary;
  if (!navigator.onLine) {
    return {
      ...primary,
      onlineMessage: "This term was not found locally. Connect to the internet to search Google.",
      onlineRetryable: true,
    };
  }

  let apiKey = getGeminiApiKey();
  if (!apiKey) apiKey = await promptForGeminiApiKey();
  if (!apiKey) {
    return {
      ...primary,
      onlineMessage: "Online dictionary lookup needs a Gemini API key stored on this device.",
      onlineNeedsSetup: true,
    };
  }

  const typedTerm = input.trim();
  result.innerHTML = `<p class="muted">Not found locally. Searching Google for <strong>${escapeHtml(typedTerm)}</strong>…</p>`;
  try {
    const online = await requestOnlineDictionaryEntry({
      term: typedTerm,
      apiKey,
      model: onlineLookupModel(),
      explainError: explainGeminiError,
    });

    // A delayed request for an old input must neither overwrite the screen nor pollute
    // the active track's additive dictionary.
    if (normalizeTerm(termInput.value) !== normalizeTerm(typedTerm)) return primary;

    if (online.status === "correction" && online.suggestedWord) {
      return {
        ...primary,
        alternatives: [
          { word: online.suggestedWord, preview: "Suggested by the grounded online lookup" },
          ...(primary.alternatives ?? []).filter(
            (item) => normalizeTerm(item.word) !== normalizeTerm(online.suggestedWord),
          ),
        ].slice(0, 6),
        onlineMessage: `Google did not verify “${typedTerm}” as an exact dictionary term. Did you mean “${online.suggestedWord}”?`,
      };
    }
    if (online.status !== "found") {
      return {
        ...primary,
        onlineMessage: "Google Search did not find enough reliable evidence to add this term automatically.",
        onlineRetryable: true,
      };
    }

    const entry = await saveOnlineDictionaryEntry(typedTerm, online);
    return userDictionaryToResult(entry, online.queryMs ?? 0);
  } catch (error) {
    const message = error instanceof OnlineDictionaryRequestError
      ? explainGeminiError(error.status, error.body)
      : (error instanceof Error ? error.message : String(error));
    return {
      ...primary,
      onlineMessage: `Online lookup failed. ${message}`,
      onlineRetryable: true,
    };
  }
}
