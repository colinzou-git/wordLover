function onlineLookupModel() {
  const current = getGeminiModel();
  return /^gemini-(?:2(?:\.0|\.5)|3(?:\.|$))/i.test(current) ? current : DEFAULT_GEMINI_MODEL;
}

async function saveOnlineDictionaryEntry(input, online) {
  const normalizedTerm = normalizeTerm(input);
  const existing = getUserDictionaryEntry(input);
  const now = nowIso();
  const entry = {
    normalizedTerm,
    word: online.canonicalWord || input.trim(),
    entryType: online.entryType || (input.includes(" ") ? "phrase" : "word"),
    phonetic: online.phonetic ?? "",
    englishMeanings: online.englishMeanings ?? [],
    chineseMeanings: online.chineseMeanings ?? [],
    partsOfSpeech: online.partsOfSpeech ?? [],
    tags: [...new Set(["online", "google-search-grounded", ...(online.tags ?? [])])],
    source: "google-search-grounded",
    definitionSource: "Google Search grounded",
    sourceUrls: online.sourceUrls ?? [],
    searchQueries: online.searchQueries ?? [],
    confidence: online.confidence ?? "medium",
    lookupProvider: "gemini-google-search",
    lookupModel: online.model ?? onlineLookupModel(),
    createdAt: existing?.createdAt ?? now,
    updatedAt: now,
    syncVersion: (existing?.syncVersion ?? 0) + 1,
  };
  userDictionaryEntries = [
    entry,
    ...userDictionaryEntries.filter((candidate) => candidate.normalizedTerm !== normalizedTerm),
  ];
  await persistUserDictionaryEntry(entry);
  return entry;
}
